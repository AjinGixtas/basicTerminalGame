[color=cyan]mkf[/color] [color=yellow][file_name][/color]
    [color=green]Creates[/color] a new file in the current or specified directory.
    [color=yellow]Arguments:[/color]
      [color=yellow][file_name][/color] – The name of the file to create. Supports paths.
    [color=green]Example:[/color]
      [color=yellow]mkf example.txt[/color] – Creates a file named `example.txt` in the current directory.
      [color=yellow]mkf folder/example.txt[/color] – Creates a file named `example.txt` in the `folder` directory.

[color=cyan]rmf[/color] [color=yellow][file_name][/color]
    [color=green]Removes[/color] a file from the current or specified directory.
    [color=yellow]Arguments:[/color]
      [color=yellow][file_name][/color] – The name of the file to remove. Supports paths.
    [color=green]Example:[/color]
      [color=yellow]rmf example.txt[/color] – Removes the file `example.txt` from the current directory.
      [color=yellow]rmf folder/example.txt[/color] – Removes the file `example.txt` from the `folder` directory.

[color=cyan]ls[/color] [color=yellow][directory][/color]
    [color=green]Lists[/color] all files and directories in the specified or current directory.
    [color=yellow]Arguments:[/color]
      [color=yellow][directory][/color] – Optional. The directory to list. Defaults to the current directory.
    [color=green]Example:[/color]
      [color=yellow]ls[/color] – Lists the contents of the current directory.
      [color=yellow]ls folder[/color] – Lists the contents of the `folder` directory.

[color=cyan]say[/color] [color=yellow][message][/color] [color=yellow][-n][/color] [color=yellow][-e][/color] [color=yellow][-r][/color]
    [color=green]Outputs[/color] the specified message to the terminal.
    [color=yellow]Arguments:[/color]
      [color=yellow][message][/color] – The message to display.
      [color=yellow]-n[/color] – Optional. Suppress the trailing newline.
      [color=yellow]-e[/color] – Optional. Enable escape sequences (e.g., `\n` for newline).
      [color=yellow]-r[/color] – Optional. Display the message in red.
    [color=green]Example:[/color]
      [color=yellow]say Hello, World![/color] – Displays `Hello, World!` in the terminal.
      [color=yellow]say -n "Hello"`[/color] – Displays `Hello` without a newline.
